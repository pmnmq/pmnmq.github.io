// server/src/main.rs
use {
    futures::stream::TryStreamExt,
    std::{collections::HashMap, sync::Arc},
};

use {
    anyhow,
    axum::{
        extract::{Path, Query, State},
        http::StatusCode,
        routing::{delete, get},
        Json, Router,
    },
    mongodb::{
        bson::{self, doc, oid::ObjectId, Bson, Document},
        options::ClientOptions,
        Client, Database,
    },
    serde::{Deserialize, Serialize},
};

const COLLECTION: &'static str = "invoices";

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let mongo_uri = std::env::var("MONGO_ADDR")
        .unwrap_or("mongodb://username:password@localhost:27017".to_string());
    let parse = ClientOptions::parse_async(mongo_uri).await;
    let client_options;
    match parse {
        Ok(options) => client_options = options,
        Err(e) => {
            panic!("Failed to parse options: {}", e);
        }
    }
    let client = Client::with_options(client_options).unwrap();
    let db = Arc::new(client.database("invoice"));
    let app = Router::new()
        .route("/api/invoices", get(get_invoices))
        .route(
            "/api/invoice",
            get(get_invoice).post(create_invoice).put(update_invoice),
        )
        .route("/api/invoice/:id", delete(delete_invoice))
        .with_state(db);

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();

    axum::serve(listener, app).await.unwrap();
    Ok(())
}

async fn get_invoices(State(db): State<Arc<Database>>) -> Json<Vec<Invoice>> {
    if let Ok(mut cursor) = db.collection::<Invoice>(COLLECTION).find(None, None).await {
        let mut result = Vec::new();
        loop {
            if let Ok(next) = cursor.try_next().await {
                match next {
                    Some(docs) => result.push(docs),
                    None => break,
                }
            } else {
                break;
            }
        }
        return Json(result);
    }
    Json(vec![])
}

async fn get_invoice(
    State(db): State<Arc<Database>>,
    Query(params): Query<HashMap<String, String>>,
) -> anyhow::Result<Json<Invoice>, (StatusCode, &'static str)> {
    if let Some(id) = params.get("id") {
        if let Ok(result) = db
            .collection::<Invoice>(COLLECTION)
            .find_one(doc! {"_id": id}, None)
            .await
        {
            if let Some(v) = result {
                return Ok(Json(v));
            }
        }
        Err((StatusCode::BAD_REQUEST, "请求错误"))
    } else {
        Err((StatusCode::BAD_REQUEST, "请求错误"))
    }
}

async fn create_invoice(
    State(db): State<Arc<Database>>,
    Json(mut data): Json<Invoice>,
) -> &'static str {
    data.id = Some(ObjectId::new().to_hex());
    if let Ok(res) = db.collection(COLLECTION).insert_one(data, None).await {
        println!("Inserted a document success, id: {}", res.inserted_id);
        return "创建成功";
    }
    "创建失败"
}

async fn update_invoice(
    State(db): State<Arc<Database>>,
    Json(data): Json<Invoice>,
) -> Result<&'static str, (StatusCode, &'static str)> {
    let data_arr: Vec<Bson> = data
        .data
        .iter()
        .map(|item| bson::to_bson(item).unwrap())
        .collect();
    match db
        .collection::<Invoice>(COLLECTION)
        .update_one(
            doc! {"_id": data.id},
            doc! {"$set": doc!{"proof_no": data.proof_no, "data": data_arr, "date": data.date}},
            None,
        )
        .await
    {
        Ok(result) => {
            println!("update success: {:?}", result.upserted_id);
            return Ok("更新成功");
        }
        Err(_) => return Err((StatusCode::BAD_REQUEST, "请求错误")),
    }
}

async fn delete_invoice(
    State(db): State<Arc<Database>>,
    Path((id,)): Path<(String,)>,
) -> Result<&'static str, (StatusCode, &'static str)> {
    match db
        .collection::<Invoice>(COLLECTION)
        .delete_one(doc! {"_id": id}, None)
        .await
    {
        Ok(result) => {
            println!("update success: {:?}", result.deleted_count);
            return Ok("删除成功");
        }
        Err(_) => return Err((StatusCode::BAD_REQUEST, "请求错误")),
    }
}

#[derive(Deserialize, Serialize, Debug)]
#[serde(rename_all = "camelCase")]
struct Invoice {
    #[serde(rename = "_id")]
    id: Option<String>,
    proof_no: String,
    date: String,
    data: Vec<InvoiceData>,
}

#[derive(Deserialize, Serialize, Debug)]
#[serde(rename_all = "camelCase")]
struct InvoiceData {
    abs: String,
    category: String,
    borrow_money: String,
    loan_money: String,
}

impl Into<Document> for InvoiceData {
    fn into(self) -> Document {
        bson::to_document(&self).unwrap()
    }
}

pub(crate) struct BoxedIntoRoute<S, E>(AxumMutex<Box<dyn ErasedIntoRoute<S, E>>>);

pub struct Route<E = Infallible>(AxumMutex<BoxCloneService<Request, Response, E>>);

enum MethodEndpoint<S, E> {
    None,
    Route(Route<E>),
    BoxedHandler(BoxedIntoRoute<S, E>),
}

#[must_use]
pub struct MethodRouter<S = (), E = Infallible> {
    get: MethodEndpoint<S, E>,
    head: MethodEndpoint<S, E>,
    delete: MethodEndpoint<S, E>,
    options: MethodEndpoint<S, E>,
    patch: MethodEndpoint<S, E>,
    post: MethodEndpoint<S, E>,
    put: MethodEndpoint<S, E>,
    trace: MethodEndpoint<S, E>,
    fallback: Fallback<S, E>,
    allow_header: AllowHeader,
}

pub struct Router<S = ()> {
    inner: Arc<RouterInner<S>>,
}

impl<S> Router<S>
where
    S: Clone + Send + Sync + 'static,
{
    #[doc = include_str!("../docs/routing/route.md")]
    #[track_caller]
    pub fn route(self, path: &str, method_router: MethodRouter<S>) -> Self {
         self.tap_inner_mut(|this| {
            panic_on_err!(this.path_router.route(path, method_router));
        })
    }
}

macro_rules! top_level_handler_fn {
    (
        $name:ident, GET
    ) => {
        top_level_handler_fn!(
            $name,
            GET
        );
    };

    (
        $name:ident, $method:ident
    ) => {
        top_level_handler_fn!(
            #[doc = concat!("Route `", stringify!($method) ,"` requests to the given handler.")]
            ///
            /// See [`get`] for an example.
            $name,
            $method
        );
    };

    (
        $(#[$m:meta])+
        $name:ident, $method:ident
    ) => {
        $(#[$m])+
        pub fn $name<H, T, S>(handler: H) -> MethodRouter<S, Infallible>
        where
            H: Handler<T, S>,
            T: 'static,
            S: Clone + Send + Sync + 'static,
        {
            on(MethodFilter::$method, handler)
        }
    };
}

top_level_handler_fn!(get, GET);
top_level_handler_fn!(delete, DELETE);
top_level_handler_fn!(post, POST);
top_level_handler_fn!(put, PUT);

top_level_handler_fn 这个声明宏看起来还是有一点复杂的，它首先匹配传入的方法是不是GET，如果是 GET 则进入第二部分，否则直接进入第三部分，在第二部分中会给函数添加文档，然后进入第三部分，第三部分中就会生成 MethodRouter，on(MethodFilter::$method, handler) 这个方法是 MethodRouter 的 on 方法

impl<S> MethodRouter<S, Infallible>
where
    S: Clone,
{
#[track_caller]
    pub fn on<H, T>(self, filter: MethodFilter, handler: H) -> Self
    where
        H: Handler<T, S>,
        T: 'static,
        S: Send + Sync + 'static,
    {
        self.on_endpoint(
            filter,
            MethodEndpoint::BoxedHandler(BoxedIntoRoute::from_handler(handler)),
        )
    }
}
