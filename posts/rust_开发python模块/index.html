#[pyclass(module = "reqrs.request", frozen)]
#[derive(Debug, Clone, Copy, Default)]
pub enum HTTPMethod {
    #[default]
    GET,
    POST,
}

impl ToPyObject for HTTPMethod {
    fn to_object(&self, py: Python) -> PyObject {
        match self {
            Self::GET => PyString::new(py, "GET").to_object(py),
            Self::POST => PyString::new(py, "POST").to_object(py),
        }
    }
}

impl From<String> for HTTPMethod {
    fn from(value: String) -> Self {
        if let Ok(http_method) = HTTPMethod::from_str(value.as_str()) {
            return http_method;
        }
        HTTPMethod::default()
    }
}

impl FromStr for HTTPMethod {
    type Err = &'static str;
    fn from_str(s: &str) -> Result<Self, Self::Err> {
        match s.to_uppercase().as_str() {
            "GET" => Ok(Self::GET),
            "POST" => Ok(Self::POST),
            _ => Err("Invalid HTTPMethod"),
        }
    }
}

pub enum PyCanonicalJSONError {
    InvalidConversion { error: String },
    PyErr { error: String },

    InvalidFloat { value: PyObject },
    InvalidCast { typename: String },
}

impl From<CanonicalJSONError> for PyCanonicalJSONError {
    fn from(error: CanonicalJSONError) -> PyCanonicalJSONError {
        PyCanonicalJSONError::InvalidConversion {

}

impl From<pyo3::PyErr> for PyCanonicalJSONError {
    fn from(error: pyo3::PyErr) -> PyCanonicalJSONError {
        PyCanonicalJSONError::PyErr {
            error: format!("{:?}", error),
        }

}

impl From<PyCanonicalJSONError> for pyo3::PyErr {
    fn from(e: PyCanonicalJSONError) -> pyo3::PyErr {
        match e {
            PyCanonicalJSONError::InvalidConversion { error } => {
                PyErr::new::<PyTypeError, _>(format!("Conversion error: {:?}", error))
            }
            PyCanonicalJSONError::PyErr { error } => {
                PyErr::new::<PyTypeError, _>(format!("Python Runtime exception: {}", error))
            }
            PyCanonicalJSONError::DictKeyNotSerializable { typename } => {
                PyErr::new::<PyTypeError, _>(format!(
                    "Dictionary key is not serializable: {}",
                    typename
                ))
            }
            PyCanonicalJSONError::InvalidFloat { value } => {
                PyErr::new::<PyTypeError, _>(format!("Invalid float: {:?}", value))
            }
            PyCanonicalJSONError::InvalidCast { typename } => {
                PyErr::new::<PyTypeError, _>(format!("Invalid type: {}", typename))
            }
        }
    }
}

            let key = if key_obj.is_none() {
                Ok("null".to_string())
            } else if let Ok(val) = key_obj.extract::<bool>() {
                Ok(if val {
                    "true".to_string()
                } else {
                    "false".to_string()
                })
            } else if let Ok(val) = key_obj.str() {
                Ok(val.to_string())
            } else {
                Err(PyCanonicalJSONError::DictKeyNotSerializable {
                    typename: key_obj
                        .to_object(py)
                        .as_ref(py)
                        .get_type()
                        .name()?
                        .to_string(),
                })
            };
            map.insert(key?, to_json(py, &value.to_object(py))?);
        }
        Ok(serde_json::Value::Object(map))
    });

    return_cast!(PyList, |x: &PyList| Ok(serde_json::Value::Array(
        x.iter()
            .map(|item| to_json(py, &item.to_object(py)).unwrap())
            .collect::<Vec<serde_json::Value>>()
    )));

    return_cast!(PyTuple, |x: &PyTuple| Ok(serde_json::Value::Array(
        x.iter()
            .map(|item| to_json(py, &item.to_object(py)).unwrap())
            .collect::<Vec<serde_json::Value>>()
    )));

    return_cast!(PyFloat, |x: &PyFloat| {
        match serde_json::Number::from_f64(x.value()) {
            Some(n) => Ok(serde_json::Value::Number(n)),
            None => Err(PyCanonicalJSONError::InvalidFloat {
                value: x.to_object(py),
            }),
        }
    });

    // At this point we can't cast it, set up the error object
    Err(PyCanonicalJSONError::InvalidCast {
        typename: obj.as_ref(py).get_type().name()?.to_string(),
    })
}

#[pyfunction]
pub fn request(
    py: Python,
    method: HTTPMethod,
    url: String,
    headers: Option<&PyDict>,
    params: Option<&PyDict>,
    proxies: Option<&PyDict>,
    timeout: Option<u64>,
    cookies: Option<&PyDict>,
    allow_redirects: Option<bool>,
    data: Option<&PyAny>,
    json: Option<&PyDict>,
) -> PyResult<String> {
    let mut base_url = Url::parse(&url).expect("invalid url");
    let mut builder = reqwest::blocking::Client::builder();
    if let Some(param_args) = params {
        let mut args = HashMap::new();
        for item in param_args.iter() {
            let (key, value) = item;
            let key_str = key.extract::<String>()?;
            let value_str = value.extract::<String>()?;
            args.insert(key_str, value_str);
        }
        let query_string = serde_urlencoded::to_string(&args).expect("parse params args error");
        base_url.set_query(Some(&query_string));
    }
    if let Some(redirect_value) = allow_redirects {
        if !redirect_value {
            builder = builder.redirect(reqwest::redirect::Policy::none());
        }
    }
    if let Some(cookie_pair) = cookies {
        let jar = reqwest::cookie::Jar::default();
        for item in cookie_pair.iter() {
            let (key, value) = item;
            let key_str = key.extract::<String>()?;
            let value_str = value.extract::<String>()?;
            jar.add_cookie_str(&format!("{}={}", key_str, value_str), &base_url);
        }
        builder = builder.cookie_provider(Arc::new(jar));
    }
    if let Some(proxy) = proxies {
        if let Some(http_proxy_url_obj) = proxy.get_item("http")? {
            let http_proxy_url = http_proxy_url_obj.extract::<String>()?;
            let proxy_url = Url::parse(&http_proxy_url).unwrap();
            let proxy = Proxy::http(format!(
                "{}://{}:{}/{}",
                proxy_url.scheme(),
                proxy_url.host_str().unwrap_or_default(),
                proxy_url.port().unwrap_or_default(),
                proxy_url.path()
            ))
            .unwrap()
            .basic_auth(
                proxy_url.username(),
                proxy_url.password().unwrap_or_default(),
            );
            builder = builder.proxy(proxy);
        }
        if let Some(http_proxy_url_obj) = proxy.get_item("https")? {
            let http_proxy_url = http_proxy_url_obj.extract::<String>()?;
            let proxy_url = Url::parse(&http_proxy_url).unwrap();
            let proxy = Proxy::https(format!(
                "{}://{}:{}/{}",
                proxy_url.scheme(),
                proxy_url.host_str().unwrap_or_default(),
                proxy_url.port().unwrap_or_default(),
                proxy_url.path()
            ))
            .unwrap()
            .basic_auth(
                proxy_url.username(),
                proxy_url.password().unwrap_or_default(),
            );
            builder = builder.proxy(proxy);
        }
    }
    if let Some(timeout_value) = timeout {
        builder = builder.timeout(time::Duration::new(timeout_value, 0))
    }

    let mut request_builder: reqwest::blocking::RequestBuilder;
    match method {
        HTTPMethod::GET => {
            request_builder = builder.build().unwrap().get(base_url);
        }
        HTTPMethod::POST => {
            request_builder = builder.build().unwrap().post(base_url);
            if let Some(data_value) = data {
                // åˆ¤æ–­æ˜¯dictè¿˜æ˜¯str
                if let Ok(dict) = data_value.downcast::<PyDict>() {
                    //åˆ¤æ–­headerä¸­Content-Type
                    if let Some(hv) = headers {
                        if let Some(content_type) = hv.get_item("Content-Type").unwrap() {
                            let content_type_str = content_type.extract::<String>()?;
                            if content_type_str.contains("json") {
                                let json_value = to_json(py, &dict.to_object(py))?;
                                request_builder = request_builder.json(&json_value);
                            } else {
                                request_builder =
                                    request_builder.form(&to_json(py, &dict.to_object(py))?);
                            }
                        } else {
                            request_builder =
                                request_builder.form(&to_json(py, &dict.to_object(py))?);
                        }
                    } else {
                        let json_value = to_json(py, &dict.to_object(py))?;
                        let s = serde_json::to_string(&json_value).unwrap();
                        println!("{}", s);
                        request_builder = request_builder.form(&json_value);
                    }
                } else if let Ok(py_str) = data_value.downcast::<PyString>() {
                    let body = py_str.extract::<String>()?;
                    request_builder = request_builder.body(body);
                }
            } else if let Some(json_data) = json {
                request_builder = request_builder.json(&to_json(py, &json_data.to_object(py))?);
            }
        }
    }

    if let Some(header_args) = headers {
        let mut header_map = header::HeaderMap::new();
        for item in header_args.iter() {
            let (key, value) = item;
            let key_str = key.extract::<String>()?;
            let value_str = value.extract::<String>()?;
            header_map.insert(
                header::HeaderName::from_str(&key_str).unwrap(),
                header::HeaderValue::from_str(&value_str).unwrap(),
            );
        }
        request_builder = request_builder.headers(header_map);
    }
    let resp = request_builder.send().unwrap();
    let content = resp.text().unwrap();
    Ok(content)
}
